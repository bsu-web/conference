CREATE DEFINER=`root`@`localhost` PROCEDURE `insert_corrals`(IN start INT, IN quant INT)
BEGIN
    DECLARE i,k INT;
    SET i=0;
    SET k=start;
    WHILE i<quant DO
         INSERT INTO `corral` (`corral_name`) VALUES (CONCAT('Загон №', CAST(k AS CHAR(10))));
         SET k=k+1;
         SET i=i+1;
    END WHILE;
END//
